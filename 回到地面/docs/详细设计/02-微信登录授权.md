## 1. 设计目标

**玩家体验**: 最小化登录阻力，一键授权即可开始游戏。开发环境自动跳过，不强制微信环境。

**关键规则**:
1. 微信小游戏环境必须走 `wx.login()` 获取授权
2. 浏览器/开发环境自动进入游客模式，不卡流程
3. 登录成功后保存本地 token，后续启动跳过此界面
4. 显示《用户协议》《隐私政策》合规弹窗

### 4.2 生命周期

```typescript
start(): void {
    // 绑定按钮事件
    if (this.wxLoginBtn) {
        this.wxLoginBtn.node.on(Button.EventType.CLICK, this._onWxLogin, this);
    }
    if (this.guestBtn) {
        this.guestBtn.node.on(Button.EventType.CLICK, this._onGuestLogin, this);
    }
    if (this.agreementLabel) {
        this.agreementLabel.node.on(Node.EventType.TOUCH_END, this._showAgreement, this);
    }

    // 首先判断环境：开发环境直接静默登录
    if (!PlatformService.instance.isWX) {
        console.log('[Login] dev mode, auto guest login');
        this._onLoginSuccess('dev_user');
    }
}
```

### 4.3 微信登录

```typescript
private async _onWxLogin(): Promise<void> {
    this._setStatus('正在登录...');
    this.wxLoginBtn && (this.wxLoginBtn.interactable = false);
    this.guestBtn && (this.guestBtn.interactable = false);

    try {
        const result = await PlatformService.instance.wxLogin();
        if (result) {
            this._onLoginSuccess(result);
        } else {
            this._setStatus('登录失败，请重试');
            this._resetButtons();
        }
    } catch (err) {
        console.error('[Login] wx login error:', err);
        this._setStatus('网络异常，请检查连接');
        this._resetButtons();
    }
}
```

### 4.4 游客登录

```typescript
private async _onGuestLogin(): Promise<void> {
    this._setStatus('正在进入...');

    // 游客模式生成临时 ID
    const guestId = 'guest_' + Date.now() + '_' + Math.floor(Math.random() * 10000);
    StorageService.instance.set('wx_openid', guestId);
    StorageService.instance.set('is_guest', 'true');

    // 延迟 0.3s 给玩家反馈
    this.scheduleOnce(() => {
        this._onLoginSuccess(guestId);
    }, 0.3);
}
```

### 4.5 登录成功跳转

```typescript
private _onLoginSuccess(openid: string): void {
    console.log('[Login] success:', openid);

    // 检查是否为首次启动（存档是否存在）
    const { PlayerDataManager } = require('../core/PlayerDataManager');
    const pdm = PlayerDataManager.getInstance();

    if (pdm.isFirstTime()) {
        // 首次 → 创建角色
        director.loadScene('create_character');
    } else {
        // 非首次 → 主城
        director.loadScene('main');
    }
}
```

### 4.6 辅助方法

```typescript
private _setStatus(msg: string): void {
    if (this.statusLabel) {
        this.statusLabel.string = msg;
    }
}

private _resetButtons(): void {
    this.wxLoginBtn && (this.wxLoginBtn.interactable = true);
    this.guestBtn && (this.guestBtn.interactable = true);
}

private _showAgreement(): void {
    if (this.agreementDetail) {
        this.agreementDetail.active = true;
        // 淡入动画
        const opacity = this.agreementDetail.getComponent(UIOpacity)
            ?? this.agreementDetail.addComponent(UIOpacity);
        opacity.opacity = 0;
        tween(opacity).to(0.2, { opacity: 255 }).start();
    }
}
```

## 5. PlatformService.ts 补充

```typescript
// 在 PlatformService 中补充完整 wxLogin 方法

import { StorageService } from '../core/StorageService';

/** 微信登录：调用 wx.login 获取 code */
async wxLogin(): Promise<string | null> {
    if (!this._isWX) {
        // 浏览器/开发环境 — 模拟登录
        console.log('[Platform] dev mode, simulation login');
        StorageService.instance.set('wx_openid', 'dev_user');
        StorageService.instance.set('is_guest', 'false');
        return 'dev_user';
    }

    try {
        // 微信小游戏环境
        const res = await new Promise<WechatMinigame.LoginRes>((resolve, reject) => {
            wx.login({
                success: resolve,
                fail: reject,
            });
        });

        if (!res.code) {
            console.warn('[Platform] wx.login returned no code');
            return null;
        }

        // TODO: 生产环境用 res.code 调后端换取 openid + session_key
        // 参考: POST https://api.weixin.qq.com/sns/jscode2session
        //       ?appid=APPID&secret=SECRET&js_code=CODE&grant_type=authorization_code
        // 返回: { openid, session_key, unionid? }

        // 开发阶段直接用 code 作为标识
        StorageService.instance.set('wx_openid', res.code);
        StorageService.instance.set('is_guest', 'false');
        return res.code;

    } catch (err) {
        console.error('[Platform] wx.login failed:', err);
        return null;
    }
}

/** 是否已登录 */
async isLoggedIn(): Promise<boolean> {
    if (!this._isWX) return true; // 开发环境默认登录
    const saved = StorageService.instance.get('wx_openid', '');
    return saved !== '';
}

/** 清除登录状态 */
clearLogin(): void {
    StorageService.instance.remove('wx_openid');
    StorageService.instance.remove('is_guest');
}

/** 是否为游客模式 */
get isGuest(): boolean {
    return StorageService.instance.get('is_guest', 'false') === 'true';
}
```

## 7. 状态管理

### 登录状态机

```
IDLE ───→ wx.login ──┬── success ──→ SAVED ──→ isFirstTime?
  │                   │                           ├── yes → create_character
  │                   └── fail ──→ RETRY           └── no  → main
  │                              (最多3次, 显示错误)
  │
  └──→ guest ──→ SAVED ──→ (同上跳转逻辑)
```

### 本地存储 key

| key | 类型 | 说明 |
|-----|------|------|
| `wx_openid` | string | 用户标识（code 或 openid） |
| `is_guest` | string | "true" 表示游客，"false" 表示微信登录 |
