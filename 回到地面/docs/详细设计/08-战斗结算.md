## 3. SettlementService

```typescript
// run/SettlementService.ts

export interface RunResult {
    isVictory: boolean;        // 通关/死亡
    characterName: string;     // 冒险者名称
    zoneName: string;          // 到达区域
    floorReached: number;      // 到达层数
    kills: number;             // 击败数
    soulStones: number;        // 获得魂石
    elapsed: number;           // 用时(秒)
}

export class SettlementService {
    /** 保存结算数据到 PlayerDataManager + 跳转主城 */
    static commit(result: RunResult): void {
        const pdm = PlayerDataManager.getInstance();
        pdm.addSoulStones(result.soulStones);
        pdm.setBestFloor(result.floorReached);
        pdm.addTotalKills(result.kills);
        pdm.addTotalRun();
        pdm.save();

        // 暂存结算数据, MainHub 读取后弹出 Panel
        StorageService.instance.set('last_run_result', JSON.stringify(result));
    }

    /** 暂存结算数据 */
    static getLastResult(): RunResult | null {
        const raw = StorageService.instance.get('last_run_result', '');
        return raw ? JSON.parse(raw) : null;
    }

    /** 清除暂存 */
    static clear(): void {
        StorageService.instance.remove('last_run_result');
    }
}
```

## 5. 数据流

```
dungeon 场景 → SettlementService.commit(result)
  → 写入 PlayerDataManager (魂石/击败/层数)
  → 暂存 result 到 StorageService
  → director.loadScene('main')

main 场景 → MainHub.onLoad()
  → 检查 SettlementService.getLastResult()
  → 不为空 → 弹出 SettlementPanel
  → 玩家操作完成后 → SettlementService.clear()
```
