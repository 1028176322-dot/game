### 5.2 VictoryUI.ts 完整代码

```typescript
import { _decorator, Component, Node, Label, Button, Sprite, Color, director } from 'cc';
import { GameManager, GameEvent } from '../core/GameManager';
import { eventBus } from '../core/TypedEventBus';
import { T } from '../core/TextManager';

const { ccclass, property } = _decorator;

@ccclass('VictoryUI')
export class VictoryUI extends Component {
    @property(Node)
    panelRoot: Node | null = null;

    @property(Label)
    titleLabel: Label | null = null;          // "🏆 恭喜通关！"

    @property(Label)
    zoneLabel: Label | null = null;           // "到达区域：熔岩火山"

    @property(Label)
    floorLabel: Label | null = null;          // "到达层数：第 12 层"

    @property(Label)
    killLabel: Label | null = null;           // "击杀数：47"

    @property(Label)
    soulStoneLabel: Label | null = null;      // "获得魂石：+1250"

    @property(Label)
    timeLabel: Label | null = null;           // "本次用时：18:32"

    @property(Button)
    doubleBtn: Button | null = null;          // 魂石翻倍广告按钮

    @property(Button)
    backBtn: Button | null = null;            // 回到营地

    private _soulStones: number = 0;
    private _floorReached: number = 0;
    private _kills: number = 0;
    private _zoneName: string = '';
    private _elapsed: number = 0;

    show(stats: {
        zoneName: string;
        floorReached: number;
        kills: number;
        soulStones: number;
        elapsed: number;
    }): void {
        this._zoneName = stats.zoneName;
        this._floorReached = stats.floorReached;
        this._kills = stats.kills;
        this._soulStones = stats.soulStones;
        this._elapsed = stats.elapsed;

        if (this.panelRoot) this.panelRoot.active = true;
        this._refresh();
    }

    hide(): void {
        if (this.panelRoot) this.panelRoot.active = false;
    }

    private _refresh(): void {
        if (this.titleLabel) this.titleLabel.string = T('victory_title');

        if (this.zoneLabel) {
            this.zoneLabel.string = T('victory_zone', { zone: this._zoneName });
        }
        if (this.floorLabel) {
            this.floorLabel.string = T('victory_floor', { floor: this._floorReached });
        }
        if (this.killLabel) {
            this.killLabel.string = T('victory_kills', { kills: this._kills });
        }
        if (this.soulStoneLabel) {
            this.soulStoneLabel.string = T('victory_soul_stones', { stones: this._soulStones });
        }
        if (this.timeLabel) {
            const mins = Math.floor(this._elapsed / 60);
            const secs = Math.floor(this._elapsed % 60);
            this.timeLabel.string = T('victory_time', { min: mins, sec: secs });
        }
    }

    onLoad(): void {
        if (this.doubleBtn) {
            this.doubleBtn.node.on(Button.EventType.CLICK, this._onDouble, this);
        }
        if (this.backBtn) {
            this.backBtn.node.on(Button.EventType.CLICK, this._onBackToMain, this);
        }
    }

    private _onDouble(): void {
        // 播放广告 → 魂石翻倍
        // 成功后 _soulStones *= 2; 刷新显示
        console.log('[Victory] double soul stones ad');
        this._soulStones *= 2;
        this._refresh();
        this.doubleBtn && (this.doubleBtn.interactable = false);
    }

    private _onBackToMain(): void {
        const gm = GameManager.instance;
        if (gm) {
            gm.commitRunResult(this._soulStones, this._floorReached, this._kills);
            gm.setPhase(GamePhase.MainMenu);
            eventBus.emit(GameEvent.DUNGEON_EXIT);
        }
        director.loadScene('main');
    }
}
```

### 5.3 RoomFlowController 改动

```typescript
// 在 onAllZonesCleared() 中打开胜利结算

onAllZonesCleared(): void {
    console.log('[RoomFlow] all zones cleared!');

    // 收集统计数据
    const gm = GameManager.instance;
    if (!gm) return;

    const stats = {
        zoneName: this._zoneName,
        floorReached: gm.currentFloor,
        kills: gm.totalKills,
        soulStones: gm.getRunSoulStones(),
        elapsed: gm.getRunElapsed(),
    };

    // 查找 VictoryUI 并显示
    const victoryUI = this.node.scene.getComponentInChildren(VictoryUI);
    if (victoryUI) {
        victoryUI.show(stats);
    }
}
```

## 7. 修复：Boss HP 条显示 (MonsterRuntimeView.ts)

```typescript
// 在 Boss 初始化后显示 HP 条

private _initHpBar(): void {
    // ... 原有逻辑 ...

    // Boss 特殊处理
    if (this._isBoss) {
        this.showHP(false); // 初始隐藏（战斗开始时显示）
    }

    // 监听 damage 事件 → Boss 受伤时显示 HP 条
    this._onDamageTaken = () => {
        if (this._isBoss) {
            this.showHP(true);
            // 5 秒后自动隐藏
            this.unschedule(this._hideBossHp);
            this.scheduleOnce(() => this.showHP(false), 5.0);
        }
    };
}
```
