## 1. 设计目标

**玩家体验**: 进入地牢前选择冒险路线，感知到"这是一次有准备、有选择的探险"。

**关键规则**:
1. 区域路线 = 3 个区域的串联（如 森林→墓穴→火山）
2. 当前解锁 1 条默认路线，其余路线需满足条件解锁
3. 默认路线 = 森林→墓穴→火山（新手→中等→困难）
4. 每个区域在 UI 上显示名称、难度、描述

### 5.1 数据定义

```typescript
interface ZoneInfo {
    id: string;
    name: string;
    difficulty: string;
    description: string;
    stars: number;           // 难度星级 (1-5)
}

interface RouteInfo {
    zones: ZoneInfo[];
    unlockCondition: string;
    isUnlocked: (pdm: PlayerDataManager) => boolean;
}

const ZONES: Record<string, ZoneInfo> = {
    forest:    { id: 'forest',    name: '翠绿森林', difficulty: '新手',   description: '温暖的森林地表',   stars: 1 },
    catacombs: { id: 'catacombs', name: '地下墓穴', difficulty: '中等',   description: '错综复杂的地下通道', stars: 3 },
    volcano:   { id: 'volcano',   name: '熔岩火山', difficulty: '困难',   description: '灼热的岩浆地带',   stars: 5 },
    swamp:     { id: 'swamp',     name: '剧毒沼泽', difficulty: '中等',   description: '瘴气弥漫的沼泽湿地', stars: 3 },
    tundra:    { id: 'tundra',    name: '永冻冰原', difficulty: '困难',   description: '极寒的冰封世界',   stars: 4 },
    abyss:     { id: 'abyss',     name: '虚空深渊', difficulty: '极难',   description: '无尽的黑暗深渊',   stars: 5 },
};

const ROUTES: RouteInfo[] = [
    {
        zones: [ZONES.forest, ZONES.catacombs, ZONES.volcano],
        unlockCondition: '',
        isUnlocked: () => true,
    },
    {
        zones: [ZONES.forest, ZONES.swamp, ZONES.tundra],
        unlockCondition: '通关翠绿森林 1 次',
        isUnlocked: (pdm) => pdm.getZoneClearCount('forest') > 0,
    },
    {
        zones: [ZONES.catacombs, ZONES.abyss, ZONES.volcano],
        unlockCondition: '通关地下墓穴 1 次',
        isUnlocked: (pdm) => pdm.getZoneClearCount('catacombs') > 0,
    },
];
```

### 5.3 显示/隐藏

```typescript
show(): void {
    if (this.panelRoot) this.panelRoot.active = true;
    this._refresh();
}

hide(): void {
    if (this.panelRoot) this.panelRoot.active = false;
}

private _refresh(): void {
    const pdm = PlayerDataManager.getInstance();
    const type = pdm.getSelectedCharacterId();
    const names: Record<string, string> = {
        warrior: '小熊·战士', archer: '小鹿·弓手',
        assassin: '小狐·刺客', mage: '小兔·法师', berserker: '小野猪·狂战士',
    };

    if (this.playerInfo) {
        this.playerInfo.string = `${names[type] ?? '冒险者'}  |  Lv${pdm.getCharacterLevel()}  魂石: ${pdm.getSoulStones()}`;
    }

    // 默认选中第一条可用路线
    this._selectedRouteIndex = 0;
    this._renderRoutes(pdm);
}
```

### 5.4 渲染路线

```typescript
private _renderRoutes(pdm: PlayerDataManager): void {
    // 当前路线卡片
    if (this.routeContainer) {
        this.routeContainer.removeAllChildren();
        const route = ROUTES[this._selectedRouteIndex];
        this._currentZones = route.zones;
        this._renderZoneChain(this.routeContainer, route.zones);
    }

    // 锁定路线列表
    if (this.lockedRouteContainer) {
        this.lockedRouteContainer.removeAllChildren();
        ROUTES.forEach((route, index) => {
            if (!route.isUnlocked(pdm) && index !== this._selectedRouteIndex) {
                this._renderLockedRoute(this.lockedRouteContainer!, route, index);
            }
        });
    }
}

private _renderZoneChain(parent: Node, zones: ZoneInfo[]): void {
    zones.forEach((zone, i) => {
        const card = new Node(`zone_${i}`);
        card.setPosition((i - 1) * 180, 0);

        // 区域卡片背景
        const bg = card.addComponent(Sprite);
        bg.color = new Color(0xF0, 0xF0, 0xF0, 0xFF);

        // 区域名
        const nameLabel = card.addComponent(Label);
        nameLabel.string = zone.name;
        nameLabel.fontSize = 16;
        nameLabel.color = new Color(0x33, 0x33, 0x33, 0xFF);
        nameLabel.position = new Vec3(0, 15, 0);

        // 难度
        const diffLabel = card.addComponent(Label);
        diffLabel.string = zone.difficulty;
        diffLabel.fontSize = 13;
        diffLabel.color = new Color(0x88, 0x88, 0x88, 0xFF);
        diffLabel.position = new Vec3(0, -10, 0);

        parent.addChild(card);

        // 箭头（非最后一个）
        if (i < zones.length - 1) {
            const arrow = new Node('arrow');
            arrow.setPosition(90, 0);
            const arrowLabel = arrow.addComponent(Label);
            arrowLabel.string = '→';
            arrowLabel.fontSize = 24;
            arrowLabel.color = new Color(0xCC, 0xCC, 0xCC, 0xFF);
            parent.addChild(arrow);
        }
    });
}

private _renderLockedRoute(parent: Node, route: RouteInfo, index: number): void {
    const node = new Node(`locked_${index}`);

    const label = node.addComponent(Label);
    const zoneNames = route.zones.map(z => z.name).join(' → ');
    label.string = `${zoneNames}    [${route.unlockCondition}]`;
    label.fontSize = 14;
    label.color = new Color(0xAA, 0xAA, 0xAA, 0xFF);

    parent.addChild(node);
}
```

### 5.5 开始冒险

```typescript
onLoad(): void {
    if (this.startBtn) {
        this.startBtn.node.on(Button.EventType.CLICK, this._onStartDungeon, this);
    }
    if (this.backBtn) {
        this.backBtn.node.on(Button.EventType.CLICK, this.hide, this);
    }
}

private _onStartDungeon(): void {
    console.log('[AreaSelect] starting route:', this._selectedRouteIndex);

    // 将选中的区域路线写入 GameManager
    const gm = GameManager.instance;
    if (gm) {
        gm.resetGame();
        gm.setSelectedZones(this._currentZones.map(z => z.id));
        gm.setPhase(GamePhase.Dungeon);
    }

    // 关闭面板，加载地牢场景
    this.hide();
    director.loadScene('dungeon');
}
```

### 5.6 GameManager 改动

// GameManager 新增
private _selectedZones: string[] = [];

setSelectedZones(zones: string[]): void {
    this._selectedZones = zones;
}

getSelectedZones(): string[] {
    return this._selectedZones;
}

// initNewRun 使用 _selectedZones 替代自动选择
initNewRun(): void {
    this.currentFloor = 1;
    this.currentPhase = GamePhase.Dungeon;
    if (this._selectedZones.length > 0) {
        this._enterZoneByList(this._selectedZones);
    } else {
        // 兜底使用默认路线
        this._selectedZones = ['forest', 'catacombs', 'volcano'];
        this._enterZoneByList(this._selectedZones);
    }
}

private _onStartAdventure(): void {
    const router = this.getComponent(UiRouter);
    if (router) {
        router.open('area_select');
    } else {
        // 兜底：直接进入地牢（如果 UiRouter 不存在）
        WXAdapter.getInstance().hideBanner();
        const gm = GameManager.instance;
        if (gm) {
            gm.resetGame();
            gm.setPhase(GamePhase.Dungeon);
            eventBus.emit(GameEvent.DUNGEON_ENTER);
        }
    }
}
