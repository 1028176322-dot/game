# Step 4: 角色选择（非首次）— 详细实现规格

> **版本**: v1.0 | **日期**: 2026-07-01 | **设计者**: GameDesigner  
> **编码**: UTF-8 | **状态**: ⚠️ 已有雏形（ShopUI 的角色 Tab），需独立

---

## 1. 设计目标

**玩家体验**: 在进入地牢前，玩家可以从已解锁的角色中选择本次冒险的伙伴，也可以在此创建新角色、解锁新角色。

**关键规则**:
1. 角色选择界面是主城的弹窗，不是独立场景
2. 显示所有已解锁角色 + 可解锁角色（标价）
3. 当前选中角色高亮显示
4. 解锁角色消耗魂石
5. 选择角色后回到主城，下次进入地牢使用该角色

## 4. CharacterSelectUI.ts 完整实现

### 4.1 角色状态模型

```typescript
interface CharacterSlot {
    id: string;                  // 角色类型 ID
    animalName: string;          // 小熊/小鹿/小狐/小兔/小野猪
    className: string;           // 战士/弓手/刺客/法师/狂战士
    unlockCost: number;          // 解锁所需魂石
    isUnlocked: boolean;         // 是否已解锁
    hasCharacter: boolean;       // 是否已创建角色（已解锁且创建）
    characterName: string | null; // 玩家自定义名称（如有）
    level: number;               // 等级
    bestFloor: number;           // 历史最高层
    totalRuns: number;           // 总局数
    isSelected: boolean;         // 当前是否选中
}

const CHARACTER_SLOTS: Omit<CharacterSlot, 'isUnlocked' | 'hasCharacter' | 'characterName' | 'level' | 'bestFloor' | 'totalRuns' | 'isSelected'>[] = [
    { id: 'warrior',   animalName: '小熊',   className: '战士',   unlockCost: 500 },
    { id: 'archer',    animalName: '小鹿',   className: '弓手',   unlockCost: 500 },
    { id: 'assassin',  animalName: '小狐',   className: '刺客',   unlockCost: 800 },
    { id: 'mage',      animalName: '小兔',   className: '法师',   unlockCost: 1000 },
    { id: 'berserker', animalName: '小野猪', className: '狂战士', unlockCost: 1200 },
];
```

private _refreshList(): void {
    const pdm = PlayerDataManager.getInstance();

    // 更新魂石显示
    if (this.soulStoneLabel) {
        this.soulStoneLabel.string = `魂石: ${pdm.getSoulStones()}`;
    }

    // 构建槽位列表
    this._buildCurrentCard(pdm);
    this._buildSlotList(pdm);
}

### 4.4 当前角色卡片

```typescript
private _buildCurrentCard(pdm: PlayerDataManager): void {
    const selected = pdm.getSelectedCharacterId();
    const slot = CHARACTER_SLOTS.find(s => s.id === selected);
    if (!slot || !this.currentCard) return;

    const name = pdm.getCharacterName() || '冒险者';
    const runs = pdm.getTotalRuns();
    const bestFloor = pdm.getBestFloor();

    if (this.currentName) {
        this.currentName.string = `${slot.animalName}·${slot.className}`;
    }
    if (this.currentInfo) {
        this.currentInfo.string = `${name}  |  最高层: ${bestFloor}`;
    }
    if (this.currentStats) {
        this.currentStats.string = `总冒险: ${runs} 次`;
    }

    // 当前角色卡片点击 → 进入地牢（快捷操作）
    this.currentCard.on(Node.EventType.TOUCH_END, () => {
        this._onStartDungeon();
    }, this);
}
```

### 4.5 角色列表

```typescript
private _buildSlotList(pdm: PlayerDataManager): void {
    if (!this.slotContainer) return;

    // 清空容器
    this.slotContainer.removeAllChildren();

    const unlocked = pdm.getUnlockedCharacterIds();
    const selected = pdm.getSelectedCharacterId();

    CHARACTER_SLOTS.forEach(slot => {
        if (slot.id === selected) return; // 跳过当前角色

        const isUnlocked = unlocked.includes(slot.id);
        const hasCreated = pdm.hasCharacterForType(slot.id);

        // 创建卡片节点
        const card = new Node(slot.id);

        // 动物头像
        const avatar = new Node('avatar');
        avatar.setPosition(-100, 0);
        // 挂 Sprite（运行时加载角色纹理）
        card.addChild(avatar);

        // 角色名+职业
        const info = card.addComponent(Label);
        info.string = `${slot.animalName}·${slot.className}`;
        info.fontSize = 16;
        info.color = new Color(0x33, 0x33, 0x33, 0xFF);
        info.position = new Vec3(-30, 15, 0);

        // 状态/操作
        const action = new Node('action');
        action.setPosition(150, 0);

        if (isUnlocked && !hasCreated) {
            // 已解锁但未创建 → [创建]
            this._attachButton(action, '创建', () => {
                this._onCreateCharacter(slot.id);
            });
        } else if (isUnlocked && hasCreated && slot.id !== selected) {
            // 已创建 → [选择]
            this._attachButton(action, '选择', () => {
                this._onSelectCharacter(slot.id);
            });
        } else if (!isUnlocked) {
            // 未解锁 → [解锁: xxx]
            const canAfford = pdm.getSoulStones() >= slot.unlockCost;
            this._attachButton(action, `解锁: ${slot.unlockCost}魂石`, () => {
                this._onUnlockCharacter(slot.id);
            }, !canAfford); // 魂石不够时禁用
        }

        card.addChild(action);

        // 锁定图标
        if (!isUnlocked) {
            const lockIcon = new Node('lock');
            lockIcon.setPosition(-100, 0);
            // 挂锁定图标 Sprite
            card.addChild(lockIcon);
        }

        this.slotContainer.addChild(card);
    });
}
```

### 4.6 操作逻辑

```typescript
/** 选择已有角色 */
private _onSelectCharacter(id: string): void {
    const pdm = PlayerDataManager.getInstance();
    pdm.selectCharacter(id);
    console.log('[CharSelect] switched to:', id);

    // 刷新界面
    this._refreshList();
}

/** 解锁新角色 */
private _onUnlockCharacter(id: string): void {
    const pdm = PlayerDataManager.getInstance();
    const slot = CHARACTER_SLOTS.find(s => s.id === id);
    if (!slot) return;

    if (pdm.getSoulStones() < slot.unlockCost) {
        console.warn('[CharSelect] not enough soul stones');
        // 可显示"魂石不足"提示
        return;
    }

    pdm.unlockCharacter(id);
    console.log('[CharSelect] unlocked:', id);

    // 刷新
    this._refreshList();
}

/** 创建角色（已解锁但未创建） */
private _onCreateCharacter(id: string): void {
    // 打开起名弹窗简化版，或直接使用默认名
    const pdm = PlayerDataManager.getInstance();
    const defaultName = `${CHARACTER_SLOTS.find(s => s.id === id)?.animalName}冒险者`;
    pdm.createAdditionalCharacter(id, defaultName);
    console.log('[CharSelect] created character for:', id);
    this._refreshList();
}

/** 进入地牢 */
private _onStartDungeon(): void {
    // 关闭面板
    this.hide();

    // 触发进入地牢
    // 由 MainUI 或 MainSceneController 监听事件后执行
    // 复用已有的 DUNGEON_ENTER 事件
    const { GameEvent } = require('../core/GameEvents');
    const { eventBus } = require('../core/TypedEventBus');
    eventBus.emit(GameEvent.DUNGEON_ENTER);
}
```

### 4.7 辅助方法

```typescript
private _attachButton(parent: Node, text: string, callback: () => void, disabled: boolean = false): void {
    const btn = new Node('btn');
    // Sprite 白底圆角
    const sprite = btn.addComponent(Sprite);
    sprite.color = disabled ? new Color(0xCC, 0xCC, 0xCC, 0xFF) : new Color(0x4A, 0x9E, 0xFF, 0xFF);

    // Label
    const label = btn.addComponent(Label);
    label.string = text;
    label.fontSize = 13;
    label.color = new Color(0xFF, 0xFF, 0xFF, 0xFF);

    // 点击
    if (!disabled) {
        btn.on(Node.EventType.TOUCH_END, callback);
    }

    parent.addChild(btn);
}
```

## 5. PlayerDataManager.ts 改动

```typescript
// 新增方法

/** 获取已解锁的角色 ID 列表 */
getUnlockedCharacterIds(): string[] {
    return this._data.unlockedCharacters;
}

/** 获取当前选中的角色 ID */
getSelectedCharacterId(): string {
    return this._data.selectedCharacter;
}

/** 切换当前角色选中 */
selectCharacter(id: string): void {
    this._data.selectedCharacter = id;
    this.save();
}

/** 解锁角色（消耗魂石） */
unlockCharacter(id: string): void {
    if (this._data.unlockedCharacters.includes(id)) return;

    const costs: Record<string, number> = {
        warrior: 500, archer: 500, assassin: 800, mage: 1000, berserker: 1200,
    };
    const cost = costs[id] ?? 0;

    if (this._data.soulStones < cost) {
        console.warn('[PlayerData] not enough soul stones');
        return;
    }

    this._data.soulStones -= cost;
    this._data.unlockedCharacters.push(id);
    this.save();
}

/** 检查某角色类型是否已有创建的角色 */
hasCharacterForType(id: string): boolean {
    // 当前单角色系统：检查 selectedCharacter 是否匹配
    return this._data.selectedCharacter === id;
}

/** 获取角色名 */
getCharacterName(): string {
    return (this._data as any).characterName ?? '';
}

/** 获取角色等级（当前固定 1） */
getCharacterLevel(): number {
    return 1; // 预留等级系统
}

/** 获取魂石数 */
getSoulStones(): number {
    return this._data.soulStones;
}

/** 获取历史最高层数 */
getBestFloor(): number {
    return this._data.bestFloor;
}

/** 获取总局数 */
getTotalRuns(): number {
    return this._data.totalRuns;
}

/** 创建额外角色（已解锁类型但未创建时调用） */
createAdditionalCharacter(typeId: string, name: string): void {
    this._data.selectedCharacter = typeId;
    (this._data as any).characterName = name;
    this.save();
}
```

## 11. 验收清单

| # | 场景 | 输入 | 预期输出 |
|---|------|------|----------|
| 1 | 打开角色选择 | 主城点"角色"按钮 | 弹出角色选择面板，显示当前角色+其他槽位 |
| 2 | 当前角色卡片 | 数据载入 | 显示角色名/职业/最高层/总局数 |
| 3 | 选择其他已创建角色 | 点击某个已创建角色的"选择"按钮 | 切换选中，卡片刷新，当前角色更新 |
| 4 | 解锁新角色 | 魂石足够，点解锁 | 魂石扣除，角色变为已解锁→可创建 |
| 5 | 解锁魂石不足 | 魂石不够，点解锁 | 按钮灰色不可点或提示不足 |
| 6 | 创建新角色 | 已解锁但未创建，点"创建" | 创建角色并设为当前 |
| 7 | 进入地牢 | 点"进入地牢" | 关闭面板，触发 DUNGEON_ENTER 事件 |
| 8 | 关闭面板 | 点"关闭" | 面板关闭，回到主城 |
