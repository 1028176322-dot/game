# 本地存储设计

> **版本**: v1.0 | **日期**: 2026-06-24
> **目标平台**: 微信小游戏 (wx.setStorageSync / wx.getStorageSync)
> **存储上限**: 10 MB（微信小游戏限制）| **实际需求**: < 100 KB

---

## 1. 存储结构

```javascript
// 根存储对象
const SAVE_KEY = 'back_to_ground_save';

// 数据结构
{
  // 元数据
  version: '0.5.0',           // 存档版本，用于兼容
  lastSaveTime: 1687612345000, // 时间戳
  
  // 魂石/经济
  soulStones: 350,            // 当前魂石总数
  totalSoulStonesEarned: 1200, // 历史累计（成就用）
  
  // 角色解锁
  characters: {                // 角色解锁状态
    warrior: true,            // 战士 - 默认解锁
    archer: false,            // 弓手
    assassin: false,          // 刺客
    mage: false,              // 法师
    berserker: false          // 狂战士
  },
  
  // 遗物池解锁
  relics: {
    bladeOfFrenzy: false,     // 狂战斧
    echoOrb: false,           // 回响之珠
    speedGauntlet: false,     // 急速手套
    ironArmor: false          // 铁甲
  },
  
  // 永久升级
  upgrades: {
    hp: 0,                    // 初始 HP+5 买了 n 次
    atk: 0,                   // 初始 ATK+1 买了 n 次
    starterItem: false        // 初始道具
  },
  
  // 天赋
  talents: {
    greed: false,             // 贪婪
    explorer: false,          // 探索者
    ironStomach: false        // 铁胃
  },
  
  // 游戏-设置
  settings: {
    musicVolume: 0.7,         // 0~1
    sfxVolume: 0.8,           // 0~1
    vibration: true,          // 震动
    adPersonalized: true      // 个性化广告推荐
  },
  
  // 统计数据
  stats: {
    totalRuns: 0,             // 总对局数
    totalKills: 0,            // 总击杀
    maxDepth: 0,              // 最深到达层数
    totalWins: 0,             // 总胜利数（通关）
    longestRunTime: 0         // 最长单局时间（秒）
  },
  
  // 每日宝箱
  dailyChest: {
    lastClaimDate: '2026-06-23',  // 上次领取日期
    streak: 3                     // 连续领取天数
  },
  
  // 缓存的上报事件（失败重试用）
  analyticsCache: []           // 最多 20 条
}
```

## 2. 字段验证

| 字段 | 类型 | 上限 | 占用 |
|------|------|------|------|
| soulStones | int | 99999 | ~8 字节 |
| characters | 5×bool | 5 个 | ~50 字节 |
| relics | 4×bool | 4 个 | ~40 字节 |
| upgrades | 3 项 | 3 | ~30 字节 |
| talents | 3×bool | 3 | ~30 字节 |
| settings | 4 项 | 4 | ~60 字节 |
| stats | 5×int | 5 | ~50 字节 |
| dailyChest | 2 项 | 2 | ~40 字节 |
| **总计** | | | **< 500 字节** |

**结论**：全部存档数据 < 500 字节，远低于 10MB 上限。即便为未来扩展预留 10 倍空间，仍低于 5KB。

## 3. 读写方案

### 3.1 保存

```javascript
// 每次关键事件后保存（不是每帧保存）

// 对局结束时
function saveAfterRun(reward) {
  let data = loadSave();
  data.soulStones += reward;
  data.stats.totalRuns += 1;
  if (reward > data.stats.maxDepth) {
    data.stats.maxDepth = reward;
  }
  data.lastSaveTime = Date.now();
  writeSave(data);
}

// 解锁时
function unlockCharacter(charId) {
  let data = loadSave();
  data.characters[charId] = true;
  writeSave(data);
}

// 保存函数
function writeSave(data) {
  data.version = '0.5.0';
  data.lastSaveTime = Date.now();
  wx.setStorageSync(SAVE_KEY, JSON.stringify(data));
}
```

### 3.2 读取

```javascript
function loadSave() {
  try {
    let raw = wx.getStorageSync(SAVE_KEY);
    let data = JSON.parse(raw);
    
    // 版本兼容
    if (!data.version || data.version !== '0.5.0') {
      return migrateSave(data); // 未来版本迁移用
    }
    
    // 给缺失的字段设置默认值（新版本新增的字段）
    return {
      version: '0.5.0',
      lastSaveTime: Date.now(),
      soulStones: 0,
      totalSoulStonesEarned: 0,
      characters: { warrior: true },
      relics: {},
      upgrades: { hp: 0, atk: 0, starterItem: false },
      talents: {},
      settings: { musicVolume: 0.7, sfxVolume: 0.8, vibration: true, adPersonalized: true },
      stats: { totalRuns: 0, totalKills: 0, maxDepth: 0, totalWins: 0, longestRunTime: 0 },
      dailyChest: { lastClaimDate: '', streak: 0 },
      analyticsCache: [],
      ...data  // 用实际数据覆盖默认值
    };
  } catch (e) {
    // 没有存档或损坏 → 返回默认
    return getDefaultSave();
  }
}

function getDefaultSave() {
  return {
    version: '0.5.0',
    lastSaveTime: Date.now(),
    soulStones: 0,
    totalSoulStonesEarned: 0,
    characters: { warrior: true },
    relics: {},
    upgrades: { hp: 0, atk: 0, starterItem: false },
    talents: {},
    settings: { musicVolume: 0.7, sfxVolume: 0.8, vibration: true, adPersonalized: true },
    stats: { totalRuns: 0, totalKills: 0, maxDepth: 0, totalWins: 0, longestRunTime: 0 },
    dailyChest: { lastClaimDate: '', streak: 0 },
    analyticsCache: []
  };
}
```

### 3.3 版本迁移（未来兼容）

```javascript
function migrateSave(oldData) {
  // 当游戏版本升级导致存档格式变化时
  // 例如从 v0.5.0 升级到 v0.6.0，新增了某个字段
  let base = getDefaultSave();
  
  // v0.5.0 → v0.6.0 迁移示例:
  if (oldData.version === '0.5.0') {
    // 合并旧数据
    base.soulStones = oldData.soulStones || 0;
    base.characters = oldData.characters || { warrior: true };
    base.relics = oldData.relics || {};
    // ...继续迁移其他字段
    
    base.version = '0.6.0'; // 更新版本号
  }
  
  return base;
}
```

## 4. 云存档方案（预留）

> 微信小游戏支持云开发（CloudBase），可作为后续扩展。
> 当前阶段本地存储足够，云存档为**预留设计**。

| 功能 | 实现方式 | 时机 |
|------|----------|------|
| 本地存档 | `wx.setStorageSync` | 每次对局结束后 |
| 云存档同步 | 调用云函数保存至数据库 | 每日首次启动时 |
| 跨设备恢复 | 用户登录后从云端拉取 | 新设备首次启动时 |

## 5. 存储安全

| 风险 | 防护措施 |
|------|----------|
| 存档损坏 | JSON.parse 失败时使用默认存档，不覆盖损坏数据 |
| 版本不兼容 | 每次保存写版本号，读取时做 version check |
| 多端覆盖 | 本地存 + 云存配合，以时间戳最新的为准 |
| 篡改数据 | 魂石等核心数值不做客户端校验（微信小游戏难以防篡改，服务端验证需要云开发）|
